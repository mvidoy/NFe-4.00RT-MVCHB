package com.md1.backend.models;

import java.io.Serializable;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.NamedQueries;
import javax.persistence.NamedQuery;
import javax.persistence.Table;
import javax.xml.bind.annotation.XmlRootElement;
import org.hibernate.annotations.DynamicUpdate;

@Entity
@Table(name = "cclasstrib_is")
@DynamicUpdate(false)
@XmlRootElement
@NamedQueries({
    @NamedQuery(name = "CCLASSTRIB_IS.findAll", query = "SELECT c FROM CCLASSTRIB_IS c ORDER BY c.CLASS_CODIGO ASC"),
    @NamedQuery(name = "CCLASSTRIB_IS.findByCodigo", query = "SELECT c FROM CCLASSTRIB_IS c WHERE c.CLASS_CODIGO = :CLASS_CODIGO")
})
public class CCLASSTRIB_IS implements Serializable {

    @Id
    private String CLASS_CODIGO;
    private String CLASS_DESCRICAO;
    private boolean CLASS_GIMPSEL;
    private boolean CLASS_VBC;
    private boolean CLASS_PIMPSELVALOR;
    private boolean CLASS_VIMPSEL;

    public String getCLASS_CODIGO() {
        return CLASS_CODIGO;
    }

    public void setCLASS_CODIGO(String CLASS_CODIGO) {
        this.CLASS_CODIGO = CLASS_CODIGO;
    }

    public String getCLASS_DESCRICAO() {
        return CLASS_DESCRICAO;
    }

    public void setCLASS_DESCRICAO(String CLASS_DESCRICAO) {
        this.CLASS_DESCRICAO = CLASS_DESCRICAO;
    }

    public boolean isCLASS_GIMPSEL() {
        return CLASS_GIMPSEL;
    }

    public void setCLASS_GIMPSEL(boolean CLASS_GIMPSEL) {
        this.CLASS_GIMPSEL = CLASS_GIMPSEL;
    }

    public boolean isCLASS_VBC() {
        return CLASS_VBC;
    }

    public void setCLASS_VBC(boolean CLASS_VBC) {
        this.CLASS_VBC = CLASS_VBC;
    }

    public boolean isCLASS_PIMPSELVALOR() {
        return CLASS_PIMPSELVALOR;
    }

    public void setCLASS_PIMPSELVALOR(boolean CLASS_PIMPSELVALOR) {
        this.CLASS_PIMPSELVALOR = CLASS_PIMPSELVALOR;
    }

    public boolean isCLASS_VIMPSEL() {
        return CLASS_VIMPSEL;
    }

    public void setCLASS_VIMPSEL(boolean CLASS_VIMPSEL) {
        this.CLASS_VIMPSEL = CLASS_VIMPSEL;
    }
}
